        // DataReceived event handler for automatic data display
        private void DataReceivedHandler(object sender, SerialDataReceivedEventArgs e)
        {
            try
            {
                SerialPort sp = (SerialPort)sender;

                int n = sp.BytesToRead;
                if (n <= 0) return;

                byte[] buf = new byte[n];
                sp.Read(buf, 0, n);

                string text = Encoding.ASCII.GetString(buf);

                lock (rxLock)
                {
                    rxBuffer.Append(text);
                }

                // optional: live stream to textBox2 as well
                this.BeginInvoke(new Action(() =>
                {
                    lineBuffer.Append(text);

                    string content = lineBuffer.ToString();
                    int newlineIndex;

                    while ((newlineIndex = content.IndexOf('\n')) >= 0)
                    {
                        // Extract one full line
                        line = content.Substring(0, newlineIndex).TrimEnd('\r');

                        // Print exactly one line
                        textBox2.AppendText(line + Environment.NewLine);

                        // Remove printed line from buffer
                        content = content.Substring(newlineIndex + 1);
                    }

                    // Keep remaining partial data
                    lineBuffer.Clear();
                    lineBuffer.Append(content);

                    textBox2.SelectionStart = textBox2.Text.Length;
                    textBox2.ScrollToCaret();
                }));

            }
            catch { }
        }

        private static string KeepOnlyHex(string s)
        {
            if (s == null) return "";
            StringBuilder sb = new StringBuilder(s.Length);
            foreach (char c in s)
            {
                if ((c >= '0' && c <= '9') ||
                    (c >= 'a' && c <= 'f') ||
                    (c >= 'A' && c <= 'F'))
                {
                    sb.Append(char.ToUpperInvariant(c));
                }
            }
            return sb.ToString();
        }

        // Convert normal string (TEXT) → HEX string (ASCII-based)
        private string StringToHex(string input)
        {
            StringBuilder sb = new StringBuilder(input.Length * 2);
            foreach (char c in input)
            {
                sb.AppendFormat("{0:X2}", (byte)c);
            }
            return sb.ToString();
        }

        // Clean HEX input (remove spaces, newlines, non-hex chars)
        private string NormalizeHex(string input)
        {
            return KeepOnlyHex(input);
        }

        private string HexToAscii_16Bytes(string hex)
        {
            hex = KeepOnlyHex(hex);

            if (hex.Length < 32)
                return "[ERR: HEX < 16 bytes]";

            StringBuilder sb = new StringBuilder();

            for (int i = 0; i < 32; i += 2)
            {
                byte b = Convert.ToByte(hex.Substring(i, 2), 16);

                // printable ASCII only
                if (b >= 32 && b <= 126)
                    sb.Append((char)b);
                else
                    sb.Append('.');
            }

            return sb.ToString();
        }

        // Wait until we have at least N hex characters received (timeout in ms)
        private string WaitForHexChars(int hexCountNeeded, int timeoutMs)
        {
            var sw = System.Diagnostics.Stopwatch.StartNew();

            while (sw.ElapsedMilliseconds < timeoutMs)
            {
                string snapshot;
                lock (rxLock)
                {
                    snapshot = rxBuffer.ToString();
                }

                string onlyHex = KeepOnlyHex(snapshot);

                if (onlyHex.Length >= hexCountNeeded)
                    return onlyHex.Substring(0, hexCountNeeded);

                System.Threading.Thread.Sleep(10);
                Application.DoEvents(); // keeps UI responsive
            }

            return null; // timeout
        }

        private List<string> SplitIntoChunks(string data, int chunkSize)
        {
            List<string> chunks = new List<string>();

            for (int i = 0; i < data.Length; i += chunkSize)
            {
                int len = Math.Min(chunkSize, data.Length - i);
                chunks.Add(data.Substring(i, len));
            }

            return chunks;
        }


        // Convert HEX string → TEXT (ASCII)
        private string HexToString(string hex)
        {
            hex = KeepOnlyHex(hex);

            StringBuilder sb = new StringBuilder();
            for (int i = 0; i + 1 < hex.Length; i += 2)
            {
                string byteStr = hex.Substring(i, 2);
                byte b = Convert.ToByte(byteStr, 16);

                // printable ASCII only (avoid garbage)
                if (b >= 32 && b <= 126)
                    sb.Append((char)b);
                else
                    sb.Append('.');
            }
            return sb.ToString();
        }

        // Thread-safe method to update textBox2
        private void SetText(string text)
        {
            if (this.textBox2.InvokeRequired)
            {
                SetTextCallback d = new SetTextCallback(SetText);
                this.Invoke(d, new object[] { text });
            }
            else
            {
                // Append the received data to textBox2
                textBox2.AppendText(text);

                // Optional: Scroll to the end
                textBox2.SelectionStart = textBox2.Text.Length;
                textBox2.ScrollToCaret();
            }
        }

        private void button2_Click(object sender, EventArgs e)
        {
            try
            {
                // Check if serial port is open
                if (!serialPort1.IsOpen)
                {
                    MessageBox.Show("Please open the serial port first.", "Port Not Open",
                        MessageBoxButtons.OK, MessageBoxIcon.Warning);
                    return;
                }

                // Check if file is selected
                if (string.IsNullOrEmpty(selectedPlainPath) || !System.IO.File.Exists(selectedPlainPath))
                {
                    MessageBox.Show("Please select a valid file first.", "No File Selected",
                        MessageBoxButtons.OK, MessageBoxIcon.Warning);
                    return;
                }

                // Read file content directly
                string fileContent = System.IO.File.ReadAllText(selectedPlainPath);

                string[] Content = new string[10];

                int blockSizeBytes = 16; // 128 bits
                int remainder = fileContent.Length % blockSizeBytes;

                if (remainder != 0)
                {
                    int paddingLength = blockSizeBytes - remainder;
                    fileContent = fileContent + new string('0', paddingLength);
                }

                int blockCount =  fileContent.Length % blockSizeBytes;

                for (int i = 0; i < blockCount; i++)
                {
                    Content[i] = fileContent.Substring(i * blockSizeBytes, blockSizeBytes);
                }


                if (string.IsNullOrWhiteSpace(fileContent))
                {
                    MessageBox.Show("Selected file is empty.", "Empty File",
                        MessageBoxButtons.OK, MessageBoxIcon.Warning);
                    return;
                }

                // Send file content over UART
                //serialPort1.Write("P");
                //serialPort1.Write(fileContent);
                //serialPort1.Write("\n");   // delimiter for Vitis

                for (int i = 0; i <= blockCount; i++)
                {


                    serialPort1.Write("P");
                    string payload;

                    if (hexMode)
                    {
                        payload = NormalizeHex(Content[i]);
                    }
                    else
                    {
                        payload = StringToHex(Content[i]);
                    }

                    serialPort1.Write(payload);
                    serialPort1.Write("\n");

                }
                textBox4.AppendText(
                    "" + fileContent.Length + " characters)\r\n");

                MessageBox.Show("File content sent successfully!", "Success",
                    MessageBoxButtons.OK, MessageBoxIcon.Information);
            }
            catch (Exception ex)
            {
                MessageBox.Show("Error sending file content: " + ex.Message, "Send Error",
                    MessageBoxButtons.OK, MessageBoxIcon.Error);
            }
        }
